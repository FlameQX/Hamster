Entity:
	User 用户类[String id, String userName, String loginName, String password, Date created, String creator, String modifier]
	UserCreation 创建用户类 [String loginName, String userName, String password]
	UserCredential 登录凭证 [String userName ,String password(加密后)]
	Page 分页[int size, int pageIndex, int total, List<T> records]

	Permission 权限 [String id, String permissionName]

	Account [String accountName, AccountType accountTypeEnum, BigDecimal balance, Date created, String creator, String modifier, Date modified, String extra [Optional<Date> statementDate, Optional<Date> RepaymentDate], Option<BigDecimal> LineOfCredit]

	BillItem 账目 [id, userid, created, accountId, costTypeEnum, amount, fromAssetsAccount]
	BillItemCreation [String userId, String accountId, Enum<CostType> costTypeEnum, BigDecimal amount]
	Bill 账单 [String id, BigDecimal expenditure, BigDecimal income, List<BillItem> billItemList]
	BillItemFilter 查询账单项过滤器 [Optional<Enum> billDate, Optional<Date> created, Optional<String> accountId, Optional<Enum> costTypeEnum]
	
	// 使用 BillItemFilter
	// StatisticsFilter 统计过滤器 [Optional<String> accountId, Optional<Enum> billDate, Optional<Enum> costTypeEnum]

Enum:
	BillDate[year/month/week/day]
	CostType[ ... ]
	AccountType
